# SDK Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

> **Note:** This plan was originally written for Fern. The implementation switched to OpenAPI Generator (`openapitools/openapi-generator-cli`) via `scripts/generate_sdks.sh`. The task steps below are a historical record; the actual file map and tooling differ.

**Goal:** Ship `@omnidapter/connect` (renamed + React hook), `@omnidapter/sdk` (TypeScript), and `omnidapter-sdk` (Python) via OpenAPI Generator, with a release workflow that generates and publishes all three.

**Architecture:** The existing `connect-sdk` package is renamed and gains a React hook as a separate subpath export. OpenAPI Generator reads `openapi/openapi.json` (exported from the running server) and generates TypeScript and Python clients. All three packages are version-bumped and published on release PR merge.

**Tech Stack:** OpenAPI Generator CLI (Docker), Vite (multi-entry lib build), `@testing-library/react` (hook tests), GitHub Actions (publish workflows), UV (Python workspace), PyPI (Python publish), npm (JS publish).

```typescript
import { useCallback, useEffect, useRef, useState } from "react";
import { OmnidapterConnect } from "./index";
import type { OmnidapterConnectOptions, OpenOptions } from "./index";

export function useOmnidapterConnect(options?: OmnidapterConnectOptions): {
    open: (options: OpenOptions) => void;
    close: () => void;
    isOpen: boolean;
} {
    const connectRef = useRef<OmnidapterConnect | null>(null);
    const [isOpen, setIsOpen] = useState(false);

    useEffect(() => {
        connectRef.current = new OmnidapterConnect(options);
        return () => {
            connectRef.current?.close();
        };
    // eslint-disable-next-line react-hooks/exhaustive-deps
    }, []);

    const open = useCallback((openOptions: OpenOptions) => {
        setIsOpen(true);
        connectRef.current?.open({
            ...openOptions,
            onSuccess: (result) => {
                setIsOpen(false);
                openOptions.onSuccess?.(result);
            },
            onError: (error) => {
                setIsOpen(false);
                openOptions.onError?.(error);
            },
            onClose: () => {
                setIsOpen(false);
                openOptions.onClose?.();
            },
        });
    }, []);

    const close = useCallback(() => {
        connectRef.current?.close();
        setIsOpen(false);
    }, []);

    return { open, close, isOpen };
}
```

```python
"""Export the omnidapter-hosted OpenAPI spec to fern/openapi/openapi.json."""
from __future__ import annotations

import json
from pathlib import Path

from omnidapter_hosted.main import create_app

OUTFILE = Path(__file__).parent.parent / "fern" / "openapi" / "openapi.json"


def main() -> None:
    app = create_app()
    spec = app.openapi()
    OUTFILE.write_text(json.dumps(spec, indent=2) + "\n")
    print(f"✓ Exported OpenAPI spec to {OUTFILE}")


if __name__ == "__main__":
    main()
```

## Task 9: Create reusable npm publish workflow

**Files:**
- Create: `.github/workflows/_publish-npm.yml`

- [ ] **Step 1: Create the workflow**

Create `.github/workflows/_publish-npm.yml`:

```yaml
name: Publish npm Package

on:
  workflow_call:
    inputs:
      package-path:
        required: true
        type: string
        description: Relative path to the package directory (e.g., client/packages/connect)
      package-name:
        required: true
        type: string
        description: npm package name for logging (e.g., @omnidapter/connect)
    secrets:
      NPM_TOKEN:
        required: true

jobs:
  publish:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          ref: ${{ github.event.pull_request.merge_commit_sha }}

      - uses: actions/setup-node@v4
        with:
          node-version: '24'
          registry-url: 'https://registry.npmjs.org'

      - name: Install dependencies
        run: |
          cd ${{ inputs.package-path }}
          npm install

      - name: Build
        run: |
          cd ${{ inputs.package-path }}
          npm run build --if-present

      - name: Check package not already published
        run: |
          PACKAGE="${{ inputs.package-name }}"
          VERSION=$(node -p "require('./${{ inputs.package-path }}/package.json').version")
          if npm view "$PACKAGE@$VERSION" version 2>/dev/null; then
            echo "❌ $PACKAGE@$VERSION already published"
            exit 1
          fi
          echo "✓ $PACKAGE@$VERSION not yet published"

      - name: Publish to npm
        run: |
          cd ${{ inputs.package-path }}
          npm publish --access public
        env:
          NODE_AUTH_TOKEN: ${{ secrets.NPM_TOKEN }}

      - name: Confirm publish
        run: echo "✓ Published ${{ inputs.package-name }}"
```

## Task 10: Create reusable PyPI publish workflow

**Files:**
- Create: `.github/workflows/_publish-pypi.yml`

- [ ] **Step 1: Create the workflow**

Create `.github/workflows/_publish-pypi.yml`:

```yaml
name: Publish PyPI Package

on:
  workflow_call:
    inputs:
      package-path:
        required: true
        type: string
        description: Relative path to the Python package directory (e.g., omnidapter-sdk)
      package-name:
        required: true
        type: string
        description: PyPI package name for logging (e.g., omnidapter-sdk)

jobs:
  publish:
    runs-on: ubuntu-latest
    permissions:
      id-token: write
    steps:
      - uses: actions/checkout@v4
        with:
          ref: ${{ github.event.pull_request.merge_commit_sha }}

      - uses: actions/setup-python@v5
        with:
          python-version: '3.12'

      - name: Install uv
        uses: astral-sh/setup-uv@v5

      - name: Build distribution
        run: |
          cd ${{ inputs.package-path }}
          uv build
          echo "✓ Built ${{ inputs.package-name }}"

      - name: Publish to PyPI
        uses: pypa/gh-action-pypi-publish@release/v1
        with:
          packages-dir: ${{ inputs.package-path }}/dist/

      - name: Confirm publish
        run: echo "✓ Published ${{ inputs.package-name }} to PyPI"
```

> Note: This uses PyPI's trusted publishing (OIDC) — no API token needed. You must configure the trusted publisher on PyPI for the `omnidapter-sdk` project (Settings → Trusted Publishers → GitHub Actions).
